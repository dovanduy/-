Sub 账号信息处理(编号)
    Rem 获取账号信息
    If 编号 > 5 Then 
        登录成功次数 = 1
        编号 = 1
    End If
	
    账号信息 = ReadUIConfig("账号" & 编号)
    WriteConfig "账号信息", 账号信息, True
    账号密码 = ReadUIConfig("密码" & 编号)
    WriteConfig "账号密码", 账号密码, True
    区服 = ReadUIConfig("区服" & 编号)
    WriteConfig "区服", 区服, True
    选服重试次数 = Int(ReadUIConfig("选服重试次数" & 编号, 10))

        Goto 获取账号信息
    End If
End Sub

Function 体力判断
    If 当前体力 = Null Then 
        TracePrint "当前体力数据异常,强制设置为0"
        ShowMessage "当前体力数据异常,强制设置为0", 2000, 50, 620
        Delay 2000
        当前体力 = 0

        体力判断 = False
    Else 
        体力判断 = True
    End If
End Function

Sub 错误记录(记录内容)
    Dim 写入行数, 文件判断结果
    文件判断结果 = Dir.Exist("localDataDir/log.txt")
    If 文件判断结果 = 1 Then 
        写入行数 = File.LinesNumber("localDataDir/log.txt") + 1
        Call File.WriteLine("localDataDir/log.txt", 写入行数, 记录内容)
        TracePrint "追加记录"
    ElseIf 文件判断结果 Then
        Call File.Write("localDataDir/log.txt", 记录内容)
        TracePrint "新建记录"
    End If
End Sub

Sub 定位主城坐标()
    ShowMessage "定位主城坐标", 1000, 50, 620
    野外主城坐标X = Int(ReadUIConfig("主城X坐标")) : 野外主城坐标Y = Int(ReadUIConfig("主城Y坐标"))
    If 野外主城坐标X <> 0 And 野外主城坐标Y <> 0 Then 
        TracePrint "用户自己设置了主城坐标,不用定位了"
        Exit Sub 
    End If
    
    Rem 定位
    Call 进出界面.回主城(1)
    //去野外
    If 进出界面.城内进固定界面(2, 相关数据解析.去野外坐标X, 相关数据解析.去野外坐标Y, 0, False) Then 
        //        Delay 2000
        容错参数 = 0
        Rem 定位坐标
        Dim ocrchar1, ocrchar2, tempStr
        ocrchar1 = 类型转换.SuperOcr(678, 597, 706, 648, "BFF3FB-111111|AFDCE2-111111|8FB0B3-111111|7F9A9C-111111|9FC6CB-111111",0.9)
        野外主城坐标X = Int(ocrchar1)
        ocrchar2 = 类型转换.SuperOcr(678, 712, 706, 765, "BFF3FB-111111|AFDCE2-111111|8FB0B3-111111|7F9A9C-111111|9FC6CB-111111",0.9)
        野外主城坐标Y = Int(ocrchar2)
        TracePrint ocrchar1, ocrchar2
        tempStr = "主城坐标" & 野外主城坐标X & "," & 野外主城坐标Y
        TracePrint tempStr
        If (野外主城坐标X = 0 Or 野外主城坐标Y = 0 Or 野外主城坐标X = Null Or 野外主城坐标Y = Null ) And 容错参数 < 5 Then 
            容错参数 = 容错参数 + 1
            TracePrint "坐标定位有问题,重试"
            Goto 定位坐标
        End If
        ShowMessage tempStr, 2000, 50, 620
        Delay 2000
        ShowMessage "如果主城坐标识别不正确,请停止运行脚本,在设置中手动输入", 1000, 50, 620
        Delay 1000
    Else 
        Goto 定位
    End If

Function 市场购买()

    For i = 0 To 7
        起始X1 = 起X1 - Int(i / 4) * X值偏移
        起始Y1 = 起Y1 + (i Mod 4) * Y值偏移
        起始X2 = 起X2 - Int(i / 4) * X值偏移
        起始Y2 = 起Y2 + (i Mod 4) * Y值偏移
        FindColor 起始X1,起始Y1,起始X2,起始Y2,"4CB786",0,1,intX,intY

                Rem 点击武将
                Touch (起始X1 + 起始X2) / 2, (起始Y1 + 起始Y2) / 2, 100
                Delay 1500
                FindPic 185,551,251,733,"Attachment:赠予.png","000000",0,0.8,intX,intY
                If intX > -1 And intY > -1 Then 
                    容错参数 = 0
                    Rem 点击赠予
                    Touch 220, 640, 100
                    Delay 1500
                    //确认结果
                    FindPic 185, 551, 251, 733, "Attachment:赠予.png", "000000", 0, 0.8, intX, intY

                        TracePrint "购买失败"

                        Goto 点击赠予
                    End If
                    If 碎片是否已购买(起始X1,起始Y1,起始X2,起始Y2) Then
                        TracePrint "购买成功"
                    ElseIf 容错参数 < 5 Then
                        容错参数 = 容错参数 + 1

                    End If
                ElseIf 容错参数 < 5 Then

                    Else

                        Goto 点击武将
                    End If
                End If
            Else

            End If
        End If
    Next	
End Function

Sub 派遣队伍()
    //类型一: 1 = 普通刷怪, 2 = 皇榜任务
    ShowMessage "派遣队伍", 1000, 50, 620
    Dim 用户设置, 打怪队伍, 队伍状态, 伤兵信息, 是否使用行军令, 打怪种类, 打怪等级, 打怪次数, 满员率设置, 队伍兵量信息, 采集队伍, 采集类型, 矿坑等级, 寻访队伍

    TracePrint "准备开始派遣队伍"
    用户设置 = 队伍设置()
    伤兵信息 = 兵事处理()
    队伍状态 = 判断队伍状态()
    打怪队伍 = 用户设置(0)
    采集队伍 = 用户设置(1)
    寻访队伍 = 用户设置(2)
    打怪种类 = 用户设置(3)
    打怪等级 = 用户设置(4)
    采集类型 = 用户设置(5)
    矿坑等级 = 用户设置(6)
    打怪次数 = Int(ReadUIConfig("打怪次数", 0))
    是否使用行军令 = ReadUIConfig("是否使用行军令", False)
    满员率设置 = Int(ReadUIConfig("满员率", 100)) 
    For i = 0 To 3
        If 打怪队伍(i) Then 
            If 队伍状态(i) = 1 Then 
                If (伤兵信息(0) / 伤兵信息(1) * 100) < Int(ReadUIConfig("伤兵比例")) Then 
                    队伍兵量信息 = 判断队伍兵量(i)
                    If 队伍兵量信息(3) > 满员率设置 Then 
                        If 打怪次数 > 派遣出征次数 Then 
                            TracePrint "出城寻找怪物"
                            Call 搜索目标(1, 打怪种类, 打怪等级)
                            Call 前往目标(1, i)
							队伍状态(i) = 0
                        End If
                    End If
                End If
            End If	
        End If
    Next
	If ReadUIConfig("自动识别矿等级") Then 
        TracePrint "开启了自动识别等级"
        矿坑等级 = Array()
        矿坑等级(识别矿坑等级() - 1) = True
    End If
    For i = 0 To 3
        If 采集队伍(i) Then 
            If 队伍状态(i) = 1 Then 
                队伍兵量信息 = 判断队伍兵量(i)
                If 队伍兵量信息(3) > 满员率设置 Then 
                    If ReadUIConfig("智能采集", True) Then 
                        采集类型 = Array()
                        采集类型(最少资源类型()) = True
                        For Each a In 采集类型
                            TracePrint a
                        Next
                    End If
                    TracePrint "出城寻找资源"
                    Call 搜索目标(2, 采集类型, 矿坑等级)
                    Call 前往目标(2, i)
					队伍状态(i) = 0
                End If
            End If	
        End If
    Next
    If ReadConfig("今日寻访剩余次数"&账号信息) = 0 Then 
        TracePrint "今日寻访次数为0"
    Else 
        For i = 0 To 3
            If 寻访队伍(i) Then 
                If 队伍状态(i) = 1 Then 
                    TracePrint "出城寻访"
                    //去野外
                    进出界面.城内进固定界面(2, 相关数据解析.去野外坐标X, 相关数据解析.去野外坐标Y, 0, False)
                    Dim 寻访坐标X, 寻访坐标Y
                    寻访坐标X = Int(ReadUIConfig("寻访城池X坐标"&登录成功次数)) : 寻访坐标Y = Int(ReadUIConfig("寻访城池Y坐标"&登录成功次数))
                    Call 移动大地图(寻访坐标X, 寻访坐标Y)
                
                    容错参数 = 0
                    Rem 点击城池
                    Call 进出界面.界面收起展开(3)
                    Touch 380, 640, 100
                    Delay 1000
                    Dim intX,intY
                    FindPic 185,666,617,1019,"Attachment:寻访图标.png","000000",0,0.7,intX,intY
                    If intX > -1 And intY > -1 Then
                        TracePrint "点击城池成功"
                        容错参数 = 0
                        Rem 点击寻访图标
                        Touch intX + 5, intY + 5, 100
                        Delay 1000
                        If 场景相关操作.进入场景是否正确(48) Then 
                            TracePrint "进入寻访界面"
                            容错参数 = 0
                            Rem 识别剩余次数
                            Dim ocrchar
                            ocrchar = 类型转换.SuperOcr(493,578,517,619,"489867-111111|5BC182-111111|3E7F57-111111",0.9)
                            TracePrint ocrchar
                            If Len(ocrchar) = 3 Then 
                                If Int(Left(ocrchar, 1)) = 0 Then 
                                    TracePrint "今日寻访次数已用完"
                                    WriteConfig "今日寻访剩余次数"&账号信息, 0, True
                                Else 
                                    Call 前往目标(3, i)
									队伍状态(i) = 0
                                End If
                            ElseIf 容错参数 < 5 Then
                                容错参数 = 容错参数 + 1
                                TracePrint "识别错误,重试"
                                Goto 识别剩余次数
                            End If
                        ElseIf 容错参数 < 5 Then
                            TracePrint "进入寻访界面失败,重试"
                            容错参数 = 容错参数 + 1
                            Goto 点击寻访图标
                        End If
                    ElseIf 容错参数 < 5 Then
                        容错参数 = 容错参数 + 1
                        TracePrint "点击城池失败,重试"
                        Goto 点击城池
                    End If
                End If	
            End If
        Next
    End If
    TracePrint "派遣循环完毕"
End Sub

Sub 搜索目标(寻找类型, 目标种类, 目标等级)
    //寻找类型,1 = 怪, 2 = 资源
    Dim 坐标增量, 新坐标X, 新坐标Y, 目标坐标信息
    目标坐标信息 = ReadConfig("目标坐标信息" & 账号信息 & 寻找类型)
    
    Call 进附近界面(寻找类型)
    //进筛选界面
    进出界面.进入子界面(46, 47, 相关数据解析.筛选界面坐标X, 相关数据解析.筛选界面坐标Y)
    Call 筛选设置(寻找类型, 目标种类, 目标等级)
    
    //目标坐标信息是为了搜索怪物从上次找到目标的地方继续,避免每次都从主城开始,浪费时间
    
    If 确认目标(寻找类型, 目标种类, 目标等级) Then 
        目标坐标信息 = Array(野外主城坐标X, 野外主城坐标Y, 1, 1, 1, 1, 1, Time())
        WriteConfig "目标坐标信息"&账号信息&寻找类型, 目标坐标信息, True
        Exit Sub
    End If
    
    //考虑到搜索资源时不需要,但是目前已经耦合到一起了,所以在最开始的时候加一个判断
    If 目标坐标信息 = Null Then 
        目标坐标信息 = Array(野外主城坐标X, 野外主城坐标Y, 1, 1, 1, 1, 1, Time())
    ElseIf (Time() - 目标坐标信息(7)) >= 7200 Or 寻找类型 = 2 Then
        TracePrint "目标坐标信息",目标坐标信息(0),目标坐标信息(1),目标坐标信息(7)
        目标坐标信息 = Array(野外主城坐标X, 野外主城坐标Y, 1, 1, 1, 1, 1, Time())
    End If
    WriteConfig "目标坐标信息" & 账号信息 & 寻找类型, 目标坐标信息, True
    
    For Each a In 目标坐标信息
        If a = Null Then 
            a = 1	
        End If
    Next
    
    
    新坐标X = Int(目标坐标信息(0)) : 新坐标Y = Int(目标坐标信息(1)) : 坐标增量 = 15
    Dim a, b, c, d
    For i = 目标坐标信息(2) To 50
        TracePrint "开始第"&i&"轮找怪"
        新坐标X = 新坐标X - 坐标增量
        新坐标Y = 新坐标Y - 坐标增量
        For a = 目标坐标信息(3) To i * 2
            TracePrint "下侧第"&a&"次找怪"
            新坐标X = 新坐标X + 坐标增量
            If 确认坐标有效性(新坐标X, 新坐标Y) Then 
                Call 移动大地图(新坐标X, 新坐标Y)
                If 确认目标(寻找类型, 目标种类, 目标等级) Then
                    目标坐标信息(0) = 新坐标X : 目标坐标信息(1) = 新坐标Y : 目标坐标信息(2) = i : 目标坐标信息(3) = a
                    WriteConfig "目标坐标信息"&账号信息&寻找类型, 目标坐标信息, True
                    Exit Sub
                End If
            End If
        Next
        a = 1
        For b = 目标坐标信息(4) To i * 2
            TracePrint "右侧第"&b&"次找怪"
            新坐标Y = 新坐标Y + 坐标增量
            If 确认坐标有效性(新坐标X, 新坐标Y) Then 
                Call 移动大地图(新坐标X, 新坐标Y)
                If 确认目标(寻找类型, 目标种类, 目标等级) Then 
                    目标坐标信息(0) = 新坐标X : 目标坐标信息(1) = 新坐标Y : 目标坐标信息(2) = i : 目标坐标信息(4) = b
                    WriteConfig "目标坐标信息"&账号信息&寻找类型, 目标坐标信息, True
                    Exit Sub
                End If
            End If
        Next
        b = 1
        For c = 目标坐标信息(5) To i * 2
            TracePrint "上侧第"&c&"次找怪"
            新坐标X = 新坐标X - 坐标增量
            If 确认坐标有效性(新坐标X, 新坐标Y) Then 
                Call 移动大地图(新坐标X, 新坐标Y)
                If 确认目标(寻找类型, 目标种类, 目标等级) Then 
                    目标坐标信息(0) = 新坐标X : 目标坐标信息(1) = 新坐标Y : 目标坐标信息(2) = i : 目标坐标信息(5) = c
                    WriteConfig "目标坐标信息"&账号信息&寻找类型, 目标坐标信息, True
                    Exit Sub
                End If
            End If
        Next
        c = 1
        For d = 目标坐标信息(6) To i * 2
            TracePrint "左侧第"&d&"次找怪"
            新坐标Y = 新坐标Y - 坐标增量
            If 确认坐标有效性(新坐标X, 新坐标Y) Then 
                Call 移动大地图(新坐标X, 新坐标Y)
                If 确认目标(寻找类型, 目标种类, 目标等级) Then 
                    目标坐标信息(0) = 新坐标X : 目标坐标信息(1) = 新坐标Y : 目标坐标信息(2) = i : 目标坐标信息(6) = d
                    WriteConfig "目标坐标信息"&账号信息&寻找类型, 目标坐标信息, True
                    Exit Sub
                End If
            End If
        Next
        d = 1
    Next
End Sub

End Function

Sub 前往目标(出征类型, 队伍)

                    Delay 2000 
                    Exit Sub
                End If
            End If
            Goto 出发
        End If

        If 出征类型 = 3 Then 
            Dim ocrchar
            ocrchar = 类型转换.SuperOcr(231,727,269,796,"B3B6B4-111111|CCCDCC-111111|DCDCDC-111111|E9E9E9-111111|7D837F-111111",1)

            End If
        End If

        Goto 进队伍界面
    End If
    TracePrint "前往目标完毕"
    场景相关操作.通用界面关闭()
    Delay 2000 
End Sub

Function 获取任务列表()
    Dim ocrchar, 任务类型,任务等级,任务子类, 任务数量, temp1, temp2,单个任务,intX,intY 
    
    //FindPic 皇榜任务坐标起X, 皇榜任务坐标初始起Y, 皇榜任务坐标止X, 皇榜任务坐标初始止Y, "Attachment:皇榜任务前往.png", "000000", 0, 0.7, intX, intY
    FindPic 440,880,550,1000,"Attachment:皇榜任务前往.png","000000",0,0.7,intX,intY
    If intX > -1 And intY > -1 Then 
        ocrchar= Ocr(438,369,506,632,"818180-111111|BABABA-111111|969696-111111|CECECE-111111|6C6C6B-111111",0.9)
        TracePrint ocrchar
			
        //判断任务类型并解析条件
        //任务类型: 1 = 打野怪, 2 = 打普通副本, 3 = 采集, 4 = 收取, 5 = 市场购买, 6 = 寻访, 7 = 打其他玩家主城, 8 = 打精英副本
        //任务数组结构: 0 = 类型, 1 = 采集数量或打怪级别或寻访次数, 2 = 子类型(野怪或者采集类型), 3 = 打怪次数, 4 = 前往图标X, 5 = 前往图标Y
        任务类型 = UTF8.Left(ocrchar, 2)
        If 任务类型 = "击杀" Then 
            temp1 = UTF8.InStr(1, ocrchar, "级")
            temp2 = UTF8.Len(ocrchar)
            任务等级 = Int(UTF8.Mid(ocrchar, 3, temp1 - 3))
            If UTF8.Mid(ocrchar, temp1 + 1, 2) = "黄巾" Then 
                任务子类 = UTF8.Mid(ocrchar, temp1 + 1, 3)
            Else 
                任务子类 = UTF8.Mid(ocrchar, temp1 + 1, 2)
            End If
        ElseIf 任务类型 = "采集" Then
            任务等级 = Null
            任务子类 = UTF8.Right(ocrchar, 2)
        ElseIf 任务类型 = "收取" Then
            任务等级 = Null
            任务子类 = Null
        ElseIf 任务类型 = "精英"
            任务子类 = Null
            任务等级 = Null
        ElseIf 任务类型 = "普通" Then
            任务子类 = Null
            任务等级 = Null
            任务数量 = UTF8.Mid(ocrchar, 7, temp1 - 7)
        End If
        单个任务 = Array(任务类型, 任务等级, 任务子类, 任务数量, intX + 5,intY + 5)
        获取任务列表 = 单个任务
        TracePrint 单个任务(0)&单个任务(1)&单个任务(2)&单个任务(3)
    Else 
        TracePrint "没找到皇榜前往图标"
    End If

End Function

Function 确认副本星级(参考坐标X, 参考坐标Y)

    If intX > -1 And intY > -1 Then
        TracePrint "副本满星"

    Else 
        FindPic 起始X1,起始Y1,起始X2,起始Y2,"Attachment:副本未通星.png","000000",0,0.97,intX,intY

Sub 进副本预览(坐标X, 坐标Y )

    If 场景相关操作.进入场景是否正确(38) Or 场景相关操作.进入场景是否正确(39) Or 场景相关操作.进入场景是否正确(3) And 容错参数 < 5 Then

                    TracePrint "次数为0,重新查找"
                    场景相关操作.通用界面关闭 
                End If

                Call 进副本预览(a(0) + 50, a(1) + 100)
                Call 获取单个副本数据(a)
                If a(3) > 0 Then 
                    挑战副本数据 = a
                    TracePrint "找到了满星3号副本,次数大于0,退出循环"
                    Exit For
                Else 
                    TracePrint "次数为0,重新查找"
                    场景相关操作.通用界面关闭()
                End If
            ElseIf a(2) = 2 And a(4) = 星级筛选条件 Then
                Call 进副本预览(a(0) + 50, a(1) + 100)
                Call 获取单个副本数据(a)
                If a(3) > 0 Then 
                    挑战副本数据 = a
                    TracePrint "找到了满星2号副本,次数大于0,退出循环"
                    Exit For
                Else 
                    TracePrint "次数为0,重新查找"
                    场景相关操作.通用界面关闭()
                End If
            ElseIf a(2) = 1 And a(4) = 星级筛选条件 Then
                Call 进副本预览(a(0) + 50, a(1) + 100)
                Call 获取单个副本数据(a)

        If ReadUIConfig("是否自动使用包子") = False Then 
            If 最终打副本次数 > 体力换算次数 Then 
                最终打副本次数 = 体力换算次数
            End If
        End If
        If 副本剩余次数 < 最终打副本次数 Then 
            最终打副本次数 = 副本剩余次数
        End If
        //如果是三星就进行扫荡,否则挑战
        TracePrint "最终打副本次数"&最终打副本次数
        Call 扫荡前装备(最终打副本次数)
    Case 3
        Rem 获取副本信息1
        挑战副本数据 = 待挑战副本()

    Case 6

                ElseIf a(2) = 目标副本编号 Then
                    Call 进副本预览(a(0) + 50, a(1) + 100)

                End If

                TracePrint "没有找到目标副本,降一关打"
                容错参数 = 容错参数 + 1

                Goto 筛选名将副本
            ElseIf 容错参数 >= 3 Then
                TracePrint "没有找到符合条件的副本数据,退出"
                ShowMessage "没有找到符合条件的副本数据,退出", 1000, 50, 620

            ElseIf 容错参数 < 5 Then

Sub 正常打副本前准备()
    容错参数 = 0
    Dim intX,intY
    TracePrint "即将进入副本选择队伍界面"
    If 进出界面.进入子界面(37, 40, 相关数据解析.挑战坐标X, 相关数据解析.挑战坐标Y) Then 
        //勾选自动配兵
        TracePrint "勾选自动配兵"
        Do Until 容错参数 > 5      
            FindPic 171,337,234,509,"Attachment:兵力勾2.png","000000",0,0.9,intX,intY
            If intX > -1 And intY > -1 Then
                TracePrint "选中了自动补兵"
                Exit Do
            Else 
                容错参数 = 容错参数 + 1
                Touch 200, 360, 100
                Delay 1000
            End If
        Loop
        Call 打副本(1, 1)
    Else 
        TracePrint "进副本队伍选择界面失败,放弃打副本"
    End If
End Sub

Sub 扫荡前装备(最终打副本次数)

    Dim intX,intY
    Rem 进入扫荡
    Touch 相关数据解析.进入扫荡坐标X, 相关数据解析.进入扫荡坐标Y, 250

Sub 建筑()
    If ReadUIConfig("是否升级建筑") = False Then 
        TracePrint "设置不升级建筑"
        Exit Sub
    End If
    TracePrint "开始建筑"
    ShowMessage "建筑", 1000, 50, 620
    Dim 建筑顺序
    建筑顺序 = ReadUIConfig("建筑优先级")
    
    If 是否有空建筑队列() Then 
        If 建筑顺序 = 0 Then 
            Call 升级大殿()
            Call 升级建筑(2)
        ElseIf 建筑顺序 = 1 Then
            Call 升级建筑(2)
        ElseIf 建筑顺序 = 2 Then
            Call 升级大殿()
        End If
    End If
    TracePrint "建筑执行完毕"
    场景相关操作.通用界面关闭()
    Delay 2000 
End Sub

Sub 沙场论剑找对手()
    TracePrint "开始沙场论剑找对手"
    Dim intX,intY,intX1,intY1,剩余挑战次数,剩余刷新次数
    Delay 3000
    Call 沙场论剑胜场奖励()
    剩余挑战次数 = 识别沙场论剑剩余次数()
    Do Until 剩余挑战次数 < 1
        For 2
            If 剩余挑战次数 < 1 Then 
                TracePrint "挑战次数不足,剩余挑战次数"
                Exit Sub
            Else  
                //1号位筛选
                FindPic 320,167,471,1004,"Attachment:沙场论剑挑战图标.png","000000",0,0.8,intX,intY
                If intX > -1 And intY > -1 Then
                    FindPic 320,167,471,1004,"Attachment:沙场论剑失败标签.png","000000",4,0.9,intX1,intY1
                    If intX1 > -1 And intY1 > -1 Then
                        TracePrint "已经挑战过该目标,但是失败了,寻找下一个"
                    Else 
                        TracePrint "准备挑战目标"
                        Call 沙场论剑挑战(intX, intY)
                    End If
                End If
                //2号位筛选
                剩余挑战次数 = 识别沙场论剑剩余次数()
                If 剩余挑战次数 < 1 Then 
                    TracePrint "挑战次数不足,剩余挑战次数"
                    Exit Sub
                Else 
                    FindPic 180,167,338,1004,"Attachment:沙场论剑挑战图标.png","000000",0,0.8,intX,intY
                    If intX > -1 And intY > -1 Then
                        FindPic 180,167,338,1004,"Attachment:沙场论剑失败标签.png","000000",4,0.9,intX1,intY1
                        If intX1 > -1 And intY1 > -1 Then
 
                        Else 
                            TracePrint "准备挑战目标"
                            Call 沙场论剑挑战(intX, intY)
                        End If
                    Else 
                        TracePrint "没找到挑战图标,这个目标已经打赢了"
                    End If
                End If
                剩余挑战次数 = 识别沙场论剑剩余次数()
                If 剩余挑战次数 < 1 Then 
                    TracePrint "挑战次数不足,剩余挑战次数"
                    Exit Sub
                Else 
                    //滑动屏幕继续
                    zm.Swipe 145, 830, 435, 830, 500
                    Delay 2000
                End If
            End If
        Next
        //3号位筛选
        剩余挑战次数 = 识别沙场论剑剩余次数()
        If 剩余挑战次数 < 1 Then 
            TracePrint "挑战次数不足,剩余挑战次数"
            Exit Sub
        Else  
                
            FindPic 130,167,290,1004,"Attachment:沙场论剑挑战图标.png","000000",0,0.8,intX,intY
            If intX > -1 And intY > -1 Then
                FindPic 130,167,290,1004,"Attachment:沙场论剑失败标签.png","000000",4,0.9,intX1,intY1
                If intX1 > -1 And intY1 > -1 Then
                    TracePrint "已经挑战过该目标,但是失败了,寻找下一个"
                Else 
                    TracePrint "准备挑战目标"
                    Call 沙场论剑挑战(intX, intY)
                End If
            Else 
                TracePrint "没找到挑战图标,这个目标已经打赢了"
            End If
        End If
        Delay 1500
        Call 沙场论剑胜场奖励()
        //刷新
        容错参数 = 0
        Rem 识别剩余刷新次数
        Dim ocrchar
        ocrchar = 类型转换.SuperOcr(535,880,558,947,"F4F4F4-111111|E8E9E9-111111|8B8E90-111111|ACAEB0-111111|BFC0C2-111111",1)
        Traceprint ocrchar
        If ocrchar <> "刷新令" Then 
            Touch 510, 945, 100
            Delay 1500
        Else 
            TracePrint "剩余刷新次数为0退出"
            Exit Do
        End If
        剩余挑战次数 = 识别沙场论剑剩余次数()	
    Loop
    TracePrint "沙场论剑挑战次数为0,退出"
End Sub

Function 识别沙场论剑剩余次数()
    容错参数 = 0
    Rem 识别剩余次数
    Dim ocrchar, temp
    ocrchar = 类型转换.SuperOcr(499,621,525,676,"F4F4F4-111111|E8E9E9-111111|8B8E90-111111|ACAEB0-111111|BFC0C2-111111",0.95)
    TracePrint ocrchar
    temp = Int(Left(ocrchar, 1))
    If temp = Null And 容错参数 < 5 Then 
        TracePrint "文本识别有误"
        容错参数 = 容错参数 + 1
        Goto 识别剩余次数
    ElseIf 容错参数 >= 5 Then
        TracePrint "错误次数过多,放弃"
        识别沙场论剑剩余次数 = 0
        Exit Function
    End If
    识别沙场论剑剩余次数 = temp
End Function

Sub 沙场论剑挑战(挑战坐标X, 挑战坐标Y)
    Dim intX,intY,intX1, intY1
    容错参数 = 0
    Rem 点击挑战
    Touch 挑战坐标X, 挑战坐标Y, 100
    Delay 1500
    FindPic 213,1017,439,1124,"Attachment:沙场论剑出击.png","000000",4,0.9,intX,intY
    If intX > -1 And intY > -1 Then
        TracePrint "找到出击按钮"
        容错参数 = 0
        Rem 点击出击
        Touch intX, intY, 100
        Delay 1500
        //检测是否战斗结束
        FindPic 53,807,178,1038,"Attachment:沙场论剑离开.png","000000",4,0.9,intX1,intY1
        If intX1 > -1 And intY1 > -1 Then 
            //胜利和败北文字出现时有动画,会有一定延迟,所以要特殊处理
            Do
                FindPic 73,110,701,742,"Attachment:沙场论剑胜利.png","000000",0,0.9,intX,intY
                If intX > -1 And intY > -1 Then 
                    TracePrint "沙场论剑挑战胜利"
                    Touch intX1, intY1, 100
                    Delay 1500
                    Exit Do
                End If
                FindPic 269,260,618,609,"Attachment:沙场论剑失败.png","000000",4,0.9,intX,intY
                If intX > -1 And intY > -1 Then 
                    TracePrint "沙场论剑挑战失败"
                    Touch intX1, intY1, 100
                    Delay 1500
                    Exit Do
                End If 
            Loop
        ElseIf 容错参数 < 3 Then
            容错参数 = 容错参数 + 1
            Delay 1000
            TracePrint "等待沙场论剑结果"
            Goto 点击出击
        End If   
    ElseIf 容错参数 < 3 Then
        容错参数 = 容错参数 + 1
        TracePrint "没有找到出击按钮,重试"
        Goto 点击挑战
    End If
End Sub

Sub 沙场论剑排名奖励()

        TracePrint "沙场论剑领奖励进入场景错误次数太多,放弃"

    End If

            FindColor 482,1079,597,1206,"2626AD",0,0.99,intX,intY

                TracePrint "有排名奖励"
                Touch intX - 5, intY - 5, 100

                    TracePrint "点击小红点领取奖励"

                    FindPic 182,707,239,891,"Attachment:沙场论剑领取奖励.png","000000",0,0.8,intX,intY

                        TracePrint "准备最终领奖"

                场景相关操作.通用界面关闭 
                Delay 2000

            场景相关操作.通用界面关闭 
            Delay 2000

End Sub
